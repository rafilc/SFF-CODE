extends CanvasLayer

func _ready():
	hide()  # start hidden so pause menu doesn't show when game loads
	pause_mode = Node.PAUSE_MODE_PROCESS  # allow ui to work even when game is paused

func open():
	show()  # show the pause menu
	get_tree().paused = true  # pause the whole game

func close():
	hide()  # hide the pause menu
	get_tree().paused = false  # unpause the game

func _on_resume_pressed():
	close()  # resume game when resume button is pressed

func _on_quit_pressed():
	get_tree().paused = false
	get_tree().change_scene("res://main_menu.tscn")  # exit the game
